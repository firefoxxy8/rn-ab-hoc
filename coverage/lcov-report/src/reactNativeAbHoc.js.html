import React from 'react';
 
const displayComponentFromVariant = (variant, props) => (
  <variant.component {...props} />
);
 
const ReactNativeAbHoc = (experiment, ...variants) => (props = { select: null }) => {
  const { select } = props;
  if (select) {
    const variant = variants.find(v => v.variant === select);
    if (!variant) {
      throw new Error(`The variant named "${select}" doesn't exist in the current experient "${experiment}"`);
    }
    return displayComponentFromVariant(variant, props);
  }
  const random = Math.random() * (variants.length - 1);
  const index = Math.round(random, 10);
  const variant = variants[index];
  return displayComponentFromVariant(variant, props);
};
 
export default ReactNativeAbHoc;
